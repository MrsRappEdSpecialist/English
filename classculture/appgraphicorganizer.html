// ========================================== // 💾 CUSTOM LOCAL STORAGE AUTO-SAVE // ========================================== const SAVE_KEY = "open_house_app_builder_v1"; function autoSave() { const data = { inputs: {}, // For text boxes with IDs radios: {}, // For radio buttons (Theme, Color, Animation) checkboxes: {}, // For feature checkboxes dynamicAnswers: {}// For the dynamic "Spill the Tea" textareas }; // 1. Save standard text inputs and textareas (Name, Period, Goals, etc.) document.querySelectorAll('input[type="text"][id], textarea[id]').forEach(el => { data.inputs[el.id] = el.value; }); // 2. Save Radio Buttons (Aesthetics, Colors, Animations) document.querySelectorAll('input[type="radio"]').forEach(el => { if (el.checked) { data.radios[el.name] = el.value; } }); // 3. Save Checkboxes (Games, Features, Accessibility) document.querySelectorAll('input[type="checkbox"].feature-checkbox').forEach(el => { data.checkboxes[el.value] = el.checked; }); // 4. Save the dynamically generated "Spill the Tea" answers document.querySelectorAll('.game-answer-input').forEach(el => { const gameName = el.getAttribute('data-game'); data.dynamicAnswers[gameName] = el.value; }); // Save to browser memory localStorage.setItem(SAVE_KEY, JSON.stringify(data)); } function loadSavedData() { const savedString = localStorage.getItem(SAVE_KEY); if (!savedString) return; // Stop if no save exists const data = JSON.parse(savedString); // 1. Restore standard inputs if (data.inputs) { for (const id in data.inputs) { const el = document.getElementById(id); if (el) el.value = data.inputs[id]; } } // 2. Restore radio buttons if (data.radios) { for (const name in data.radios) { const el = document.querySelector(`input[type="radio"][name="${name}"][value="${data.radios[name]}"]`); if (el) el.checked = true; } } // 3. Restore checkboxes if (data.checkboxes) { for (const val in data.checkboxes) { const el = document.querySelector(`input[type="checkbox"][value="${val}"]`); if (el) el.checked = data.checkboxes[val]; } // IMPORTANT: Trigger your existing function to rebuild the "Spill the Tea" fields based on loaded checkboxes! if (typeof updateAnswerFields === 'function') { updateAnswerFields(); } } // 4. Restore "Spill the Tea" answers // We use a tiny timeout (50ms) to wait for updateAnswerFields() to finish creating the boxes first setTimeout(() => { if (data.dynamicAnswers) { for (const gameName in data.dynamicAnswers) { const el = document.querySelector(`.game-answer-input[data-game="${gameName}"]`); if (el) el.value = data.dynamicAnswers[gameName]; } } }, 50); } // Trigger the save whenever the user types or clicks anything document.addEventListener('input', autoSave); document.addEventListener('change', autoSave); // Load the data the moment the page opens window.addEventListener('DOMContentLoaded', loadSavedData);
